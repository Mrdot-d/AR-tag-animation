  <model-viewer id="model"
                src="https://modelviewer.dev/shared-assets/models/RobotExpressive.glb"

  </model-viewer>

  <div style="text-align:center; margin-top:10px;">
    <button id="playAnim">Avvia Animazione</button>
    <button id="stopAnim">Ferma Animazione</button>
  </div>

  <script>
    const model = document.getElementById('model');

    document.getElementById('playAnim').addEventListener('click', () => {
      // avvia l'animazione 'Dance'
      model.playAnimation('Dance');
    });

    document.getElementById('stopAnim').addEventListener('click', () => {
      model.pause(); // ferma tutte le animazioni
    });
  </script>
